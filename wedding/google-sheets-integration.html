Google Sheets Integration for Wedding RSVPs

Step 1: Create a Google Sheet

  1. Go to Google Sheets
  2. Create a new spreadsheet
  3. Add these headers in row 1:
    • A1: Name
    • B1: Phone
    • C1: Method
    • D1: Attendance
    • E1: Transport
    • F1: Notes
    • G1: Date

Step 2: Set up Google Apps Script

  1. In your Google Sheet, go to Extensions > Apps Script
  2. Replace the code with this:

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  
  sheet.appendRow([
    data.name,
    data.phone,
    data.method,
    data.attendance,
    data.transport || '',
    data.notes || '',
    new Date()
  ]);
  
  return ContentService.createTextOutput(JSON.stringify({status: 'success'}))
    .setMimeType(ContentService.MimeType.JSON);
}

Step 3: Deploy the Script

  1. Click "Deploy" > "New deployment"
  2. Choose "Web app"
  3. Set access to "Anyone"
  4. Copy the Web App URL

Step 4: Update Your Website

Replace the localStorage code in script.js with this:


// Replace localStorage with Google Sheets
async function submitRSVP(rsvpData) {
  const GOOGLE_SHEETS_URL = 'YOUR_WEB_APP_URL_HERE';
  
  try {
    const response = await fetch(GOOGLE_SHEETS_URL, {
      method: 'POST',
      body: JSON.stringify(rsvpData),
      headers: {
        'Content-Type': 'application/json'
      }
    });
    
    if (response.ok) {
      console.log('RSVP submitted successfully');
      return true;
    } else {
      console.error('Failed to submit RSVP');
      return false;
    }
  } catch (error) {
    console.error('Error submitting RSVP:', error);
    return false;
  }
}
